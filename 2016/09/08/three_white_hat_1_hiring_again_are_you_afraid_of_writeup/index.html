第二個坑 SQL注入

漏洞來源：

http://www.wooyun.org/bugs/wooyun-2010-064792
http://www.wooyun.org/bugs/wooyun-2010-0171945

waf.php

$uri = explode("?",$_SERVER['REQUEST_URI']);
if(isset($uri[1])) {
    $parameter = explode("&",$uri[1]);
    foreach ($parameter as $k => $v) {
        $v1 = explode("=",$v);
        if (isset($v1[1])) {
            $_REQUEST[$v1[0]] = stripStr($v1[1]);
        }
    }
}
